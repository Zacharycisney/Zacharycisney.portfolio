                                driver = webdriver.Chrome()
                    
                                # Open the first tab
                                driver.get("https://robinhood.com")
                    
                                # Open additional tabs
                                driver.execute_script("window.open('https://robinhood.com/stocks/WMT')")
                                driver.execute_script("window.open('https://robinhood.com/stocks/SPY')")
                                driver.execute_script("window.open('https://robinhood.com/stocks/MSFT')")
                                driver.execute_script("window.open('https://robinhood.com/stocks/AXP')")
                    
                                # Store the handles of all opened tabs
                                handles = driver.window_handles
                    
                                # Iterate through the tabs and switch between them every 20 seconds
                                current_tab = 0
                                while True:
                                    # Switch to the next tab
                                    driver.switch_to.window(handles[current_tab])
                    
                                    # Wait for 20 seconds
                                    time.sleep(20)
                    
                                    # Move to the next tab
                                    current_tab = (current_tab - 1) % len(handles)

Tree Parser

Fall 2021
              
              // Copyright 2021, Bradley Peterson, Weber State University, all rights reserved. (11/2021)
              
              #include <iostream>
              #include <string>
              #include <stack>
              #include <sstream>
              #include <cmath>
              #include <cstdio>
              
              using std::stack;
              using std::istringstream;
              using std::ostringstream;
              using std::string;
              using std::cout;
              using std::cerr;
              using std::cin;
              using std::endl;
              using std::pow;
              
              
              class Node {
              public:
                  string mathString;
                  Node* llink = nullptr;
                  Node* rlink = nullptr;
              };
              
              class TreeParser {
              private:
                  stack<string> mathStack;
                  double castStrToDouble(string const& s);
                  string castDoubleToStr(const double d);
                  void initialize();
              
                  bool isDigit(char c);
                  bool isOperator(char c);
                  void processExpression(Node* p);//raw ptr
                  void computeAnswer(Node* ptr);
              
              protected:
                  string expression;
                  int position;
              
              public: 
                  TreeParser();
                  void displayParseTree();
                  void processExpression(string &expression);
                  string computeAnswer();
                  void inOrderTraversal(Node* ptr);
                  void postOrderTraversal(Node* ptr);
              
                  Node* root = nullptr;
              
              };
              
              
              bool TreeParser::isDigit(char c) {
                  
                  if (isdigit(c)) { //does this work?
                      return true;
                  }
              
                  else {
                      return false;
                  }
              }
              
              
              bool TreeParser::isOperator(char c) {
                  if (c == '+' || c == '-' || c == '*' || c == '/' || c == '^') {
                      return true;
                  }
                  
                  else {
                      return false;
                  }
              }
              
              
              void TreeParser::processExpression(string& expression) {
                  
                  if (expression != "") {//if empty
                  
                      this->expression = expression;
                      position = 0;
                      root = new Node();
              
                      processExpression(root);
                  }
                  return;
              }
              
              
              void TreeParser::processExpression(Node* p) {
              
                  string tempString = "";
              
                  while (position < (expression.size())) {
              
                      if (expression.at(position) == '(') {
                          Node* temp = new Node();
                          p->llink = temp;
                          position++;
                          processExpression(p->llink);
                      }
                      
                      else if (isDigit(expression.at(position)) == true || expression.at(position) == '.') {
                          tempString = expression.at(position);
                          cout << tempString;
                          position++;
                          while (isDigit(expression.at(position)) == true || expression.at(position) == '.') {
                              
                              tempString += expression.at(position);
                              position++;
                          
                          }
                          p->mathString = tempString;
                          return;
                      }
                      else if (isOperator(expression.at(position)) == true) {
                          Node* temp = new Node();
                          p->rlink = temp;
                          p->mathString = expression.at(position);
                          position++;
                          processExpression(p->rlink);
                      }
                      else if (expression.at(position) == ')') {
                          position++;
                          return;
                      }
                      else if (expression.at(position) == ' ') {
                          position++;
                      }
                      tempString = "";
                  }
              
              }
              
              
              string TreeParser::computeAnswer() {
                  computeAnswer(root);
                  return mathStack.top();
              }
              
              
              void TreeParser::computeAnswer(Node* ptr) {
                  if (ptr) {
                      computeAnswer(ptr->llink);
                      computeAnswer(ptr->rlink);
              
              
                      if (isDigit(ptr->mathString.at(0)) == true) {
                          mathStack.push(ptr->mathString);
                      }
              
                      else if (isOperator(ptr->mathString.at(0)) == true) {
                          double secondNum = castStrToDouble(mathStack.top());
                          mathStack.pop();
                          double firstNum = castStrToDouble(mathStack.top());
                          mathStack.pop();
              
                          if (ptr->mathString == "+") {
                              mathStack.push(castDoubleToStr(firstNum + secondNum));
                          }
                          else if (ptr->mathString.at(0) == '-') {
                              mathStack.push(castDoubleToStr(firstNum - secondNum));
                          }
                          else if (ptr->mathString.at(0) == '*') {
                              mathStack.push(castDoubleToStr(firstNum * secondNum));
                          }
                          else if (ptr->mathString.at(0) == '/') {
                              mathStack.push(castDoubleToStr(firstNum / secondNum));
                          }
                          else if (ptr->mathString.at(0) == '^') {
                              mathStack.push(castDoubleToStr((pow(firstNum, secondNum))));
                          }
              
                      }
              
                  }
              }
              
              
              void TreeParser::inOrderTraversal(Node* ptr) {
                  if (ptr) {
                      inOrderTraversal(ptr->llink); // Left 
                      cout << ptr->mathString << " "; //Act
                      inOrderTraversal(ptr->rlink); // Right
                  }
              }
              
              void TreeParser::postOrderTraversal(Node* ptr) {
                  if (ptr) {
                      postOrderTraversal(ptr->llink); // Left 
                      postOrderTraversal(ptr->rlink); // Right
                      cout << ptr->mathString << " "; //Act
                  }
              }
              
              //pre written
              void TreeParser::initialize() {
                  expression = "";
                  position = 0;
                  while (!mathStack.empty()) {
                      mathStack.pop();
                  }
              }
              
              double TreeParser::castStrToDouble(const string& s) {
                  istringstream iss(s);
                  double x;
                  iss >> x;
                  return x;
              }
              
              string TreeParser::castDoubleToStr(const double d) {
                  ostringstream oss;
                  oss << d;
                  return oss.str();
              
              }
              
              TreeParser::TreeParser() {
                  initialize();
              }
              
              
              void TreeParser::displayParseTree() {
                  cout << "The expression seen using in-order traversal: ";
                  inOrderTraversal(root);
                  cout << endl;
                  cout << "The expression seen using post-order traversal: ";
                  postOrderTraversal(root);
                  cout << endl;
              }
              
              
              int main() {
                  TreeParser tp;
                  string expression;
                  cout << "Please enter an expression: ";
                  getline(cin, expression);
                  tp.processExpression(expression);
                  tp.displayParseTree();
                  cout << "The answer to the expression is: " << tp.computeAnswer() << endl;
                  return 0;
              }

Joins and Sub-queries

                              1 - Dates -- MM/DD/YYYY unless otherwise specified.

                              2a - alphabetic listing -- means sort it by last name, then first name

                              4 - My preferred contact is: scott.sonderegger@gmail.com

                              6 - If you are asked to sort on a column, don't use DESC unless you are specifically asked to do so.

                              /*1. List the student ID, name and employer of the student that has enrolled the most times.
                              Arrange the list in alphabetical order of last name then first name.
                              */
                              SELECT
                                  s.student_id,
                                  s.first_name,
                                  s.last_name,
                                  s.employer
                              FROM
                                  student s
                              WHERE
                                  s.student_id IN (SELECT 
                                                  student_id
                                              FROM
                                                  enrollment
                                              GROUP BY
                                                  student_id
                                              HAVING
                                              COUNT(section_id) = (SELECT
                                                  MAX(COUNT(se.section_id))
                                              FROM
                                                  student s JOIN enrollment e
                                                  ON (s.student_id = e.student_id)
                                                  JOIN section se
                                                  ON (e.section_id = se.section_id)
                                              GROUP BY
                                                  s.student_id)
                                                  )
                              ORDER BY
                                  3,
                                  4;
                              /*3. Provide a list of names and cities of students and instructors that live in zipcode 10025. Identify each person's role as either "Student" or "Instructor". Sort the list by role, last name and first name.
                              */
                              SELECT
                                  'student' as Role,
                                  s.first_name,
                                  s.last_name,
                                  z.city
                              FROM
                                  student s JOIN zipcode z
                                  ON(s.zip = z.zip)
                              WHERE
                                  z.zip = 10025
                              UNION
                              SELECT
                                  'instructor',
                                  i.first_name,
                                  i.last_name,
                                  z.city
                              FROM
                                  instructor i JOIN zipcode z
                                  ON(i.zip = z.zip)
                              WHERE
                                  z.zip = 10025
                              ORDER BY
                                  1,
                                  3,
                                  2
                              /*4. Create a query that lists location, number of sections taught in that location and number of students enrolled in sections at that location. Sort by location.
                              */
                              SELECT
                                  s.location,
                                  COUNT(DISTINCT(s.section_id)) as Sections_taught,
                                  COUNT(e.student_id) as Student
                              FROM
                                  section s JOIN enrollment e
                                  ON (s.section_id = e.section_id)
                              GROUP BY
                                  location
                              ORDER BY
                                  location;
                              
                              /*5. Create a query that shows all of the individual grades for student ID 112 in section 95 and also the average of those grades. The individual grades should come first with the average at the bottom. List the grade type code and numeric grade. The average row should have a caption of, "Average for student 112 is".
                              */
                              SELECT
                                  grade_type_code,
                                  numeric_grade
                              FROM
                                  grade
                              WHERE
                                  student_id = 112
                                  AND section_id = 95
                              UNION ALL
                              SELECT
                                  'Average for student 112 is',
                                  ROUND(AVG(numeric_grade), 2)
                              FROM    
                                  grade
                              WHERE
                                  student_id = 112
                                  AND section_id = 95
                              ORDER BY 
                                  1 DESC
                              /*6. Create an alphabetic listing of all instructors and the number of sections that they have taught.
                              */
                              SELECT    
                                  i.first_name,
                                  i.last_name,
                                  count(s.section_id) as Sections_Taught
                              FROM
                                  instructor i LEFT OUTER JOIN section s
                                  ON (i.instructor_id = s.instructor_id)
                              GROUP BY
                                  i.first_name,
                                  i.last_name
                              ORDER BY
                                  i.first_name,
                                  i.last_name
                              /*7. List the course number and description of courses wherein students have received grades for all possible grade types. Order by Course Number.
                              */

                                  c.course_no,

                              WHERE
                                  grade_type_code = 'MT'
